Calls by site 

calls_by_site_plot <- function(df) {
  require(dplyr)
  df %>%
    filter(., !is.na(site_id)) %>%
    ggplot(.) +
    aes(fct_infreq(site_id), fill = site_id) +
    geom_bar() +
    theme(axis.text.x = element_text(
      angle = 90,
      vjust = 0.5,
      hjust = 1
    )) + # Rotate text
    labs(x = "site") +
    theme(legend.position = "none")
}

calls_by_site_plot(screen_df)
